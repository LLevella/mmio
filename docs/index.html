mmio

mmio is a header-only C++17 library for reading, writing, and converting Matrix Market matrices.

Install

cmake -S . -B build -DCMAKE_BUILD_TYPE=Release
cmake --build build
cmake --install build --prefix install

CMake Consumer

find_package(mmio REQUIRED)
target_link_libraries(app PRIVATE mmio::mmio)

Example

#include "mmio.h"

#include <sstream>

int main() {
  std::istringstream input(
      "%%MatrixMarket matrix coordinate real general\n"
      "3 3 2\n"
      "1 1 10\n"
      "3 2 -5\n");

  auto csr = mmio::ReadCSRMatrixMarket<double>(input);
  return csr.nnz() == 2 ? 0 : 1;
}
